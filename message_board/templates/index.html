from flask import Flask, request, jsonify from flask_sqlalchemy import SQLAlchemy import os app = Flask(__name__) # PostgreSQL 接続（環境変数から URL を取得） DATABASE_URL = os.environ.get("DATABASE_URL") app.config["SQLALCHEMY_DATABASE_URI"] = DATABASE_URL app.config["SQLALCHEMY_TRACK_MODIFICATIONS"] = False db = SQLAlchemy(app) # モデル定義 class Post(db.Model): id = db.Column(db.Integer, primary_key=True) name = db.Column(db.String(80)) content = db.Column(db.String(200)) # 投稿作成 @app.route("/post", methods=["POST"]) def create_post(): data = request.json new_post = Post(name=data["name"], content=data["content"]) db.session.add(new_post) db.session.commit() return jsonify({"message": "Post created"}), 201 # 投稿一覧取得 @app.route("/posts", methods=["GET"]) def get_posts(): posts = Post.query.all() return jsonify([ {"id": p.id, "name": p.name, "content": p.content} for p in posts ])
